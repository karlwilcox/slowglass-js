description
    Traffic demo
    All images created by Affinty App generative AI
    call from URL /scripts/vehicles.txt
end description

scene road_users
    begin
        load /assets/vehicles/vehicles.png as vehicles
        make near_reduction be 75
        make near_roadheight be 380
        make far_reduction be 80
        make far_roadheight be 360
        make counter be 1
        # row 1
        make v_refusetruck be 20 46 1172 516
        make v_firetruck be 1242 76 1374 480
        make v_minivan be 2632 214 868 322
        # row 2
        make v_citybus be 4 610 1524 506
        make v_greycar be 1686 748 894 318
        make v_bluecar be 2596 768 850 288
        # row 3
        make v_darkcar be 48 1196 970 324
        make v_taxi be 1262 1184 924 336
        make v_minibus be 2298 1084 1072 438
        # row 4
        make v_van be 78 1558 1072 454
        make v_sportscar be 1242 1726 1026 290
        make v_greencar be 2364 1652 984 352
        make run be yes
        make elapsed_ltor be $ELAPSED
        make elapsed_rtol be $ELAPSED

    every 1 seconds
        if $run
            if $PERCENT > 50
                match traffic to v_ at start from $VARIABLES
                choose vehicle from $traffic
                choose direction from _ltor _rtol
                make gap be elapsed$direction
                if ($ELAPSED - $$gap) > 3
                    clone drive$direction as drive$counter
                    start scene drive$counter $$vehicle
                    increment counter
                    make $gap be $ELAPSED
                endif
            endif
        endif
end scene

scene drive_rtol
    begin
        sprite create vehicle from road_users:vehicles view $PARAMS
        shrink vehicle by ${road_users:near_reduction}
        place vehicle at -200 330 500
        align vehicle bottom to ${road_users:near_roadheight}
        move vehicle horizontally to 1000 in (8 + ($CHANCE * 4)) seconds
    after 13 seconds
        stop $SCENENAME
        remove scene $SCENENAME
end scene

scene drive_ltor
    begin
        sprite create vehicle from road_users:vehicles view $PARAMS
        shrink vehicle by ${road_users:far_reduction}
        flip vehicle horizontally
        place vehicle at 1000 330 400
        align vehicle bottom to ${road_users:far_roadheight}
        move vehicle horizontally to -200 in (8 + ($CHANCE * 4)) seconds
    after 13 seconds
        stop $SCENENAME
        remove scene $SCENENAME
end scene
