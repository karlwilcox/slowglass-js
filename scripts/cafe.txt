# Cafe at night, slightly animated (after Van Gogh)

script height 1920
script width 1080

stage size 540 960

begin

    # Load all our resources
    load /demo/cafe/cafe.png
    # cafe scene from https://commons.wikimedia.org/ Van Gogh Cafe Terrace At Night
    sprite create star1 load from /demo/cafe/star1.png
    sprite create star2 load from /demo/cafe/star2.png
    sprite create star3 load from /demo/cafe/star3.png
    sprite create lamp load from /demo/cafe/lamp.png
    then

    # Place them
    put cafe as background
    place star1 at 747 376 201
    place star2 at 881 60 202
    place star3 at 346 11 203
    place lamp at 367 1014 204

    # Twinkle those stars
    flicker star1
    flicker star2
    flicker star3

    # Jiggle that candle
    flicker lamp with chance 30
    jiggle lamp 10 10 with chance 40
