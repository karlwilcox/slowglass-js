# Fireworks demo
# All images created by Affinty App generative AI
# call from URL /slow-glass/scripts/fireworks.txt

display width 800
display height 450
gravity 50

begin
    # Create our stage...
    with /slow-glass/assets/fireworks
    load starry-sky.png as starry-sky
    use starry-sky as background
    load crowd-of-people.png as crowd
    use crowd as foreground
    load tree1.png as left-tree
    load tree2.png as right-tree
    sprite create left-tree from left-tree
    sprite create right-tree from right-tree
    darken left-tree to 70
    darken right-tree to 70
    place left-tree at 20 200 depth 250
    place right-tree at 780 200 depth  251

===> for test in one two three
===> log test is now $test
===> next

end file

every 3 seconds
    wait (1 + (4 * $CHANCE)) seconds
    start firework1

scene firework1
   begin
        with /slow-glass/assets/fireworks
        load firework1-1.png as firework1
        make launch-site be (600 * $CHANCE)
        # launch from anywhere between 100 and 600 but tend to point towards the centre
        make start-angle be ((((${launch-site} / 600) * 50) - 25) * -1)
        make launch-angle be (${start-angle} + (-20 + (40 * $CHANCE)))
        make launch-speed be (180 + (180 * $CHANCE))
        make big-size be (100 + $PERCENT)
        choose tint from red lightcyan lawngreen yellow orangered gold tomato
        sprite create firework1 from firework1
        tint firework1 $tint
        rotate firework1 (350 * $CHANCE)
        resize firework1 to 10 10
        resize firework1 to ${big-size} ${big-size} in 3 seconds
        place firework1 at (100 + ${launch-site}) 450 depth 220
        # jiggle firework1 10 10 100
        launch firework1 at ${launch-angle} with force ${launch-speed}
    after 1 second
        darken _MAIN_:left-tree to 50 in 1 second
        darken _MAIN_:right-tree to 50 in 1 second
    after 3 seconds
        darken _MAIN_:left-tree to 70 in 1 second
        darken _MAIN_:right-tree to 70 in 1 second
        darken firework1 to 90 in 1 second
    after 4 seconds
        fade firework1 to 0 in 1 second
    after 7 seconds
        remove sprite firework1
        stop scene
end scene
